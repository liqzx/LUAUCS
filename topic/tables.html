Tables

Array-like Tables

-- Creating an array local weapons = {"Sword", "Bow", "Axe"} -- Accessing elements (1-based indexing) print(weapons[1]) -- Output: Sword -- Adding elements weapons[4] = "Dagger" table.insert(weapons, "Staff") -- Adds to end -- Iterating for i, weapon in ipairs(weapons) do print(i .. ": " .. weapon) end

Dictionary Tables

-- Creating a dictionary local player = { name = "Builder", health = 100, level = 5, inventory = {"Sword", "Potion"} } -- Accessing values print(player.name) -- Output: Builder print(player["health"]) -- Alternate syntax -- Modifying values player.health = player.health - 20 player.gold = 500 -- Adds new key -- Iterating for key, value in pairs(player) do print(key .. ": " .. tostring(value)) end

Table Methods

local items = {"Sword", "Shield", "Potion"} -- table.insert (add element) table.insert(items, "Key") -- Adds to end table.insert(items, 2, "Armor") -- Inserts at position 2 -- table.remove (remove element) table.remove(items, 1) -- Removes first item -- table.sort (sort elements) table.sort(items) -- Alphabetical order -- table.concat (join strings) local joined = table.concat(items, ", ") print(joined) -- Output: Armor, Potion, Shield, Key
