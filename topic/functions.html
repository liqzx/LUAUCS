Functions

Basic Function

-- Function definition local function greetPlayer(playerName) return "Hello, " .. playerName .. "!" end -- Function call local message = greetPlayer("Builder") print(message) -- Output: Hello, Builder!

Multiple Parameters

local function calculateDamage(baseDamage, isCritical) if isCritical then return baseDamage * 2 else return baseDamage end end local damage = calculateDamage(50, true) print("Damage dealt: " .. damage) -- Output: Damage dealt: 100

Anonymous Functions

-- Assigning to a variable local sayGoodbye = function(name) return "Goodbye, " .. name end -- As a callback task.delay(2, function() print("This runs after 2 seconds") end)

Table Methods

-- Creating a player object local player = { name = "Builder", health = 100, -- Method definition takeDamage = function(self, amount) self.health = self.health - amount print(self.name .. " took " .. amount .. " damage!") end } -- Calling the method player:takeDamage(25) -- Output: Builder took 25 damage!
