Control Flow

If Statements

local health = 75 if health < 20 then print("Low health warning!") elseif health < 50 then print("Health is getting low") else print("Health is good") end

While Loops

-- Countdown timer local count = 5 while count > 0 do print("Countdown: " .. count) count -= 1 wait(1) -- Wait 1 second end print("Blast off!")

For Loops

-- Numeric for loop for i = 1, 5 do print("Spawning enemy #" .. i) end -- With step value for i = 10, 0, -2 do print(i) end

Repeat Until

-- Game loop example local gameOver = false local playerHealth = 100 repeat -- Game logic here playerHealth -= 10 print("Player health: " .. playerHealth) if playerHealth <= 0 then gameOver = true end wait(1) until gameOver print("Game over!")
